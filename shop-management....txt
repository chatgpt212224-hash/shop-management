<script>
  let products = [];

  function loadData() {
    const saved = localStorage.getItem('shop-products');
    if (saved) products = JSON.parse(saved);
  }
  function saveData() {
    localStorage.setItem('shop-products', JSON.stringify(products));
  }

  function addProduct() {
    const name = document.getElementById('name').value.trim();
    const price = parseFloat(document.getElementById('price').value) || 0;
    const cost = parseFloat(document.getElementById('cost').value) || 0;
    const qty = parseInt(document.getElementById('qty').value) || 0;
    if (!name || price <= 0) {
      alert('กรุณากรอกชื่อและราคาสินค้า');
      return;
    }
    products.push({ id: Date.now(), name, price, cost, qty });
    saveData();
    render();
    ['name','price','cost','qty'].forEach(id => document.getElementById(id).value = '');
    document.getElementById('name').focus();
  }

  function deleteProduct(id) {
    if (!confirm('ลบสินค้านี้?')) return;
    products = products.filter(p => p.id !== id);
    saveData();
    render();
  }

  function getStockBadge(qty) {
    if (qty === 0) return '<span class="stock-badge stock-out">หมด</span>';
    if (qty < 10) return `<span class="stock-badge stock-low">${qty} (ใกล้หมด)</span>`;
    return `<span class="stock-badge stock-ok">${qty}</span>`;
  }

  function render() {
    const tbody = document.getElementById('tbody');
    if (products.length === 0) {
      tbody.innerHTML = '<tr><td colspan="7" class="empty">ยังไม่มีสินค้า · เริ่มต้นเพิ่มสินค้าด้านบนได้เลย</td></tr>';
    } else {
      tbody.innerHTML = products.map((p, i) => `
        <tr>
          <td>${String(i+1).padStart(2,'0')}</td>
          <td><strong>${p.name}</strong></td>
          <td class="price">฿${p.price.toLocaleString()}</td>
          <td>฿${p.cost.toLocaleString()}</td>
          <td class="price" style="color: var(--green)">+฿${(p.price - p.cost).toLocaleString()}</td>
          <td>${getStockBadge(p.qty)}</td>
          <td><button class="danger" onclick="deleteProduct(${p.id})">ลบ</button></td>
        </tr>
      `).join('');
    }

    document.getElementById('totalItems').textContent = products.length;
    const totalQty = products.reduce((s,p) => s + p.qty, 0);
    const totalValue = products.reduce((s,p) => s + p.price * p.qty, 0);
    const lowStock = products.filter(p => p.qty > 0 && p.qty < 10).length;
    document.getElementById('totalQty').textContent = totalQty.toLocaleString();
    document.getElementById('totalValue').textContent = totalValue.toLocaleString();
    document.getElementById('lowStock').textContent = lowStock;
  }

  function showDate() {
    const d = new Date();
    const months = ['ม.ค.','ก.พ.','มี.ค.','เม.ย.','พ.ค.','มิ.ย.','ก.ค.','ส.ค.','ก.ย.','ต.ค.','พ.ย.','ธ.ค.'];
    document.getElementById('today').textContent =
      `${d.getDate()} ${months[d.getMonth()]} ${d.getFullYear() + 543}`;
  }

  loadData();
  showDate();
  render();
</script>
